description: >
  <p>Calculates new <code>x</code>, <code>y</code>, and <code>z</code>
  components that are proportionally the

  same distance between two vectors. The <code>amt</code> parameter is the
  amount to

  interpolate between the old vector and the new vector. 0.0 keeps all

  components equal to the old vector's, 0.5 is halfway between, and 1.0 sets

  all components equal to the new vector's.</p>

  <p>The static version of <code>lerp()</code>, as in <code>p5.Vector.lerp(v0,
  v1, 0.5)</code>,

  returns a new <a href="#/p5.Vector">p5.Vector</a> object and doesn't change

  the original.</p>

    function draw() {
      background(200);

      let v0 = createVector(50, 50);
      let v1 = createVector(30, 0);
      let v2 = createVector(0, 30);
      let v3 = p5.Vector.lerp(v1, v2, 0.5);

      drawArrow(v0, v1, 'red');
      drawArrow(v0, v2, 'blue');
      drawArrow(v0, v3, 'purple');

      describe('Three arrows extend from the center of a gray square. A red arrow points to the right, a blue arrow points down, and a purple arrow points to the bottom right.');
    }

    function drawArrow(base, vec, myColor) {
      push();
      stroke(myColor);
      strokeWeight(3);
      fill(myColor);
      translate(base.x, base.y);
      line(0, 0, vec.x, vec.y);
      rotate(vec.heading());
      let arrowSize = 7;
      translate(vec.mag() - arrowSize, 0);
      triangle(0, arrowSize / 2, 0, -arrowSize / 2, arrowSize, 0);
      pop();
    }
