description: >
  <p>Divides a vector's <code>x</code>, <code>y</code>, and <code>z</code>
  components by the same number,

  separate numbers, the components of another

  <a href="#/p5.Vector">p5.Vector</a> object, or an array of numbers. Calling

  <code>div()</code> with no arguments has no effect.</p>

  <p>The static version of <code>div()</code>, as in <code>p5.Vector.div(v,
  2)</code>, returns a new

  <a href="#/p5.Vector">p5.Vector</a> object and doesn't change the

  originals.</p>

      background(200);

    function drawArrow(base, vec, myColor) {
      push();
      stroke(myColor);
      strokeWeight(3);
      fill(myColor);
      translate(base.x, base.y);
      line(0, 0, vec.x, vec.y);
      rotate(vec.heading());
      let arrowSize = 7;
      translate(vec.mag() - arrowSize, 0);
      triangle(0, arrowSize / 2, 0, -arrowSize / 2, arrowSize, 0);
      pop();
    }
