  By default, p5.js loops through draw() continuously, executing the code within

  it. However, the <a href="#/p5/draw">draw()</a> loop may be stopped by calling

  <a href="#/p5/noLoop">noLoop()</a>. In that case, the <a
  href="#/p5/draw">draw()</a>

  loop can be resumed with loop().Avoid calling loop() from inside setup().Use
  <a href="#/p5/isLooping">isLooping()</a> to check the current state of <a
  href="#/p5/loop">loop()</a>.

    let x = 0;

      createCanvas(100, 100);
      noLoop();

    function draw() {
      background(204);
      x = x + 0.1;
      if (x > width) {
        x = 0;
      }
      line(x, 0, x, height);
    }

    function mousePressed() {
      loop();
    }

    function mouseReleased() {
      noLoop();
    }
