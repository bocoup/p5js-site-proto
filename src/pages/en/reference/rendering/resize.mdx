      createCanvas(100, 100, WEBGL);

    function mouseMoved() {
      framebuffer.resize(
        max(20, mouseX),
        max(20, mouseY)
      );
    }

    function draw() {
      // Draw to the framebuffer
      framebuffer.begin();

      sphere(20);
