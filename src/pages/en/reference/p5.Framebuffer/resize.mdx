    function mouseMoved() {
      framebuffer.resize(
        max(20, mouseX),
        max(20, mouseY)
      );
    }

    function draw() {
      // Draw to the framebuffer
      framebuffer.begin();
      background(255);
      normalMaterial();
      sphere(20);
