  <a href="#/p5/ambientLight">ambientLight()</a>,

    <div>
    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);
      describe(
        'Three white spheres. Each appears as a different color due to lighting.'
      );
    }
    function draw() {
      background(200);
      noStroke();

      ambientLight(255, 0, 0);
      translate(-30, 0, 0);
      ambientMaterial(255);
      sphere(13);

      noLights();
      translate(30, 0, 0);
      ambientMaterial(255);
      sphere(13);

      ambientLight(0, 255, 0);
      translate(30, 0, 0);
      ambientMaterial(255);
      sphere(13);
    }
