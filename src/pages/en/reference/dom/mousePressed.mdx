descriptionText: >-
  Calls a function when the mouse is pressed over the element.

  Calling `myElement.mousePressed(false)` disables the function.Note: Some
  mobile browsers may also trigger this event when the element

  receives a quick tap.
params: []
kind: function
name: mousePressed
memberof: p5.Element
namespace: mousePressed
examples:
  - |-
    <div>
    <code>
    function setup() {
      // Create a canvas element and
      // assign it to cnv.
      let cnv = createCanvas(100, 100);

      background(200);

      // Call randomColor() when the canvas
      // is pressed.
      cnv.mousePressed(randomColor);

      describe('A gray square changes color when the mouse is pressed.');
    }

    // Paint the background either
    // red, yellow, blue, or green.
    function randomColor() {
      let c = random(['red', 'yellow', 'blue', 'green']);
      background(c);
    }
    </code>
    </div>
