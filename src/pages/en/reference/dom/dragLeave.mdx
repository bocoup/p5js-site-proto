      background(200);

      // Call byeFile() when a
      // file is dragged over,
      // then off the canvas.
      cnv.dragLeave(byeFile);

      describe('A gray square. The text "bye, file" appears when a file is dragged over, then off the square.');
    }

      text('bye, file', 50, 50);
