descriptionText: >-
  The <a href="#/p5/touchEnded">touchEnded()</a> function is called every time a
  touch ends. If no

  <a href="#/p5/touchEnded">touchEnded()</a> function is defined, the <a
  href="#/p5/mouseReleased">mouseReleased()</a> function will be

  called instead if it is defined.<br><br>

  Browsers may have different default behaviors attached to various touch

  events. To prevent any default behavior for this event, add "return false"

  to the end of the method.
params:
  - name: event
    description: optional TouchEvent callback argument.
    type: TouchEvent

    // Release touch within the image to
    // change the value of the rectangle

    let value = 0;
    function draw() {
      fill(value);
      rect(25, 25, 50, 50);
      describe('50-by-50 black rect turns white with touch.');
    }
    function touchEnded() {
      if (value === 0) {
        value = 255;
      } else {
        value = 0;
      }
    }
    </code>
