description: >
  <p>Sets the width of the stroke used for lines, points, and the border around

  shapes. All widths are set in units of pixels.</p>

  <p>Note that <code>strokeWeight()</code> is affected by any transformation or
  scaling that

  has been applied previously.</p>
line: 314
params:
  - name: weight
    description: |
      <p>the weight of the stroke (in pixels).</p>
    type: Number
itemtype: method
class: p5
example:
  - >-

    <div>

    <code>

    // Default.

    line(20, 20, 80, 20);

    // Thicker.

    strokeWeight(4);

    line(20, 40, 80, 40);

    // Beastly.

    strokeWeight(10);

    line(20, 70, 80, 70);

    describe('Three horizontal black lines. The top line is thin, the middle is
    medium, and the bottom is thick.');

    </code>

    </div>


    <div>

    <code>

    // Default.

    line(20, 20, 80, 20);

    // Adding scale transformation.

    scale(5);

    // Coordinates adjusted for scaling.

    line(4, 8, 16, 8);

    describe('Two horizontal black lines. The top line is thin and the bottom is
    five times thicker than the top.');

    </code>

    </div>
