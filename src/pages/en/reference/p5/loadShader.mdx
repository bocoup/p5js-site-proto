description: >
  <p>Creates a new <a href="#/p5.Shader">p5.Shader</a> object

  from the provided vertex and fragment shader files.</p>

  <p>The shader files are loaded asynchronously in the

  background, so this method should be used in <a
  href="#/p5/preload">preload()</a>.</p>

  <p>Shaders can alter the positioning of shapes drawn with them.

  To ensure consistency in rendering, it's recommended to use the vertex shader
  in the <a href="#/p5/createShader">createShader example</a>.</p>

  <p>Note, shaders can only be used in WEBGL mode.</p>
line: 12

    description: |
      <p>path to file containing vertex shader
      source code</p>
    type: String
  - name: fragFilename
    description: |
      <p>path to file containing fragment shader
      source code</p>
    type: String
  - name: callback
    description: >
      <p>callback to be executed after loadShader

      completes. On success, the p5.Shader object is passed as the first
      argument.</p>
    type: Function
    optional: true
  - name: errorCallback
    description: |
      <p>callback to be executed when an error
      occurs inside loadShader. On error, the error is passed as the first
      argument.</p>
    type: Function
    optional: true

example:
  - |-

    <div modernizr='webgl'>
    <code>
    let mandel;
    function preload() {
      // load the shader definitions from files
      mandel = loadShader('assets/shader.vert', 'assets/shader.frag');
    }
    function setup() {
      createCanvas(100, 100, WEBGL);
      // use the shader
      shader(mandel);
      noStroke();
      mandel.setUniform('p', [-0.74364388703, 0.13182590421]);
      describe('zooming Mandelbrot set. a colorful, infinitely detailed fractal.');
    }

    function draw() {
      mandel.setUniform('r', 1.5 * exp(-6.5 * (1 + sin(millis() / 2000))));
      quad(-1, -1, 1, -1, 1, 1, -1, 1);
    }
    </code>
    </div>
