description: >
  <p>The inverse of <a href="#/p5/sin">sin()</a>, returns the arc sine of a

  value. This function expects input values in the range of -1 to 1. By

  default, <code>asin()</code> returns values in the range -π ÷ 2

  (about -1.57) to π ÷ 2 (about 1.57). If the

  <a href="#/p5/angleMode">angleMode()</a> is <code>DEGREES</code> then values
  are

  returned in the range -90 to 90.</p>
line: 60

  - name: value
    description: |
      <p>value whose arc sine is to be returned.</p>
    type: Number
itemtype: method
class: p5
example:
  - |-

    <div>
    <code>
    let a = PI / 3;
    let s = sin(a);
    let as = asin(s);
    text(`${round(a, 3)}`, 35, 25);
    text(`${round(s, 3)}`, 35, 50);
    text(`${round(as, 3)}`, 35, 75);

    describe('The numbers 1.047, 0.866, and 1.047 written on separate rows.');
    </code>
    </div>

    <div>
    <code>
    let a = PI + PI / 3;
    let s = sin(a);
    let as = asin(s);
    text(`${round(a, 3)}`, 35, 25);
    text(`${round(s, 3)}`, 35, 50);
    text(`${round(as, 3)}`, 35, 75);

    describe('The numbers 4.189, -0.866, and -1.047 written on separate rows.');
    </code>
    </div>

  description: arc sine of the given value.
  type: Number
