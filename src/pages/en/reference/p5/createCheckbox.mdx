description: >
  <p>Creates a checkbox <code>&lt;input&gt;&lt;/input&gt;</code> element.
  Checkboxes extend

  the <a href="#/p5.Element">p5.Element</a> class with a <code>checked()</code>
  method.

  Calling <code>myBox.checked()</code> returns <code>true</code> if it the box
  is checked and

  <code>false</code> otherwise.</p>

  <p>The first parameter, <code>label</code>, is optional. It's a string that
  sets the label

  to display next to the checkbox.</p>

  <p>The second parameter, <code>value</code>, is also optional. It's a boolean
  that sets the

  checkbox's value.</p>

    type: String
    optional: true
  - name: value
    description: >
      <p>value of the checkbox. Checked is <code>true</code> and unchecked is
      <code>false</code>.</p>

    optional: true

example:
  - |-

    <div>
    <code>
    let checkbox;

    function setup() {
      // Create a checkbox and place it beneath the canvas.
      checkbox = createCheckbox();
      checkbox.position(0, 100);

      describe('A black square with a checkbox beneath it. The square turns white when the box is checked.');
    }

    function draw() {
      // Use the checkbox to set the background color.
      if (checkbox.checked()) {
        background(255);
      } else {
        background(0);
      }
    }
    </code>
