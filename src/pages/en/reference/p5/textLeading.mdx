description: >
  <p>Sets the spacing between lines of text when

  <a href="#/p5/text">text()</a> is called. Spacing is measured in pixels.</p>

  <p>Calling <code>textLeading()</code> without an argument returns the current
  spacing.</p>
line: 85
itemtype: method
class: p5
example:
  - >-

    <div>

    <code>

    // "\n" starts a new line of text.

    let lines = 'one\ntwo';


    text(lines, 10, 25);


    textLeading(30);

    text(lines, 70, 25);


    describe('The words "one" and "two" written on separate lines twice. The
    words on the left have less vertical spacing than the words on the right.');

    </code>

    </div>
