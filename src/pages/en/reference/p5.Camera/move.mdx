    let cam;

      normalMaterial();
      cam = createCamera();

    function draw() {
      background(200);

      // move the camera along its local axes
      cam.move(delta, delta, 0);

      // every 100 frames, switch direction
      if (frameCount % 150 === 0) {
        delta *= -1;
